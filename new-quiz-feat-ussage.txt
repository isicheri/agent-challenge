  answers: Array<{ questionId: string; selectedOptionId: string | null }>,
  timeTaken: number

    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ attemptId, answers, timeTaken }),
  });
  
  const result = await response.json();
  return result;
};
